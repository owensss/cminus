int gcd(int a,int b)
{
	if(b==0)
		return a;
	return gcd(b,a-a/b*b);
}
int max(int a[],int n)
{
	int i;
	int m;
	i=0;
	m=0;
	while(i<n)
	{
		if(m<a[i])
			m=a[i];
		i=i+1;
	}
	return m;
}
